
### database:
database hiki;


###tables:



table users (
    username varchar(50) not null unique,
    id varchar(50) primary key,
    password varchar(255) not null,
    phone varchar(15) not null,
    created_at timestamp default current_timestamp,
    city varchar(15) not null references adresses(id)
)
table adresses (
	id varchar(15) primary key,
    lat float,
    lon float
)
table posts (
	id int auto_increment primary key,
    finder varchar(50) not null references users(id),
    item varchar(50),
    created_at timestamp default current_timestamp,
    place char, #this will reference the camping places' table that brouhi takes care of (can be changed to a reference to the place)
    phone varchar(15) not null,
    picture varchar(50)
)

table trails (
    id int auto_increment primary key,
    name varchar(100) not null,
    level varchar(20) not null,
    min_age int default 0,
    max_group_size int,
    video_url varchar(255),
    tips text,
    checklist text,
    details text,
    created_at timestamp default current_timestamp
)

table trail_reviews (
    id int auto_increment primary key,
    trail_id int not null references trails(id),
    user_id varchar(50) not null references users(id),
    rating int not null,
    comment text,
    created_at timestamp default current_timestamp
)
